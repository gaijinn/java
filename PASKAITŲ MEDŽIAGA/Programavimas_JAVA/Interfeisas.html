package com.company; 
 
import java.awt.*; 
 
// Interfeisas Grafinis objektas 
interface GraphicObject{ 
    void draw(Graphics graphics); 
    void resize(int scale); 
    void moveTo(int x, int y); 
} 
// Klasė kvadratas 
 
class Square implements GraphicObject { 
    int bound; 
 
    @Override 
    public void draw(Graphics graphics) { 
        Polygon polygon = new Polygon(); 
        polygon.addPoint(bound, bound); 
        graphics.drawPolygon(polygon); 
        System.out.println("Kvadratas sėkmingai nupieštas"); 
    } 
 
    @Override 
    public void resize(int scale) { 
        int result = getSquareSize() * scale; 
        System.out.println(result); 
    } 
 
    @Override 
    public void moveTo(int x, int y) { 
        moveSquare(x, y); 
    } 
 
    int getSquareSize() { 
        int size = 10; 
        for (int i = 0; i < 10; i++) { 
            size += i; 
        } 
        return size; 
    } 
 
    void moveSquare(int x, int y) { 
        System.out.println("Kvadratas sėkmingai perkeltas: " + x + " " + y); 
    } 
} 
    // Klasė Tiesė 
class Line implements GraphicObject{ 
    int x1,x2,y1,y2; 
    @Override 
    public void draw(Graphics graphics) { 
        graphics.drawLine(x1,y1,x2,y2); 
    } 
    @Override 
    public void resize(int scale) { 
        System.out.println("Mastelis sėkmingai pakeistas"+scale); 
    } 
    @Override 
    public void moveTo(int x, int y) { 
        System.out.println("Tiesė sėkmingai perkelta į: x"+x+" y:"+y); 
    } 
} 
class Oval implements GraphicObject{ 
 
    int x, y, width, height; 
 
    @Override 
    public void draw(Graphics graphics) { 
        graphics.drawOval(x, y, width, height); 
    } 
    @Override 
    public void resize(int scale) { 
        System.out.println("Mastelis sėkmingai pakeistas"+scale); 
    } 
 
    @Override 
    public void moveTo(int x, int y) { 
        System.out.println("Ovalas sėkmingai perkeltas į: x"+x+" y:"+y); 
    } 
} 
 
public class Main { 
    public static void main(String[] args) { 
 
        GraphicObject graphicObject = new Line(); 
 
        graphicObject.moveTo(10,10); 
 
        graphicObject = new Square(); 
 
        graphicObject.moveTo(20,20); 
 
        graphicObject = new Oval(); 
 
        graphicObject.moveTo(30, 50); 
    } 
}
