29   class Person {

37           this.age = age;

39           this.address = address; 
40           this.lastname = lastname;

46       void pushArrayToList(Person [] person){ 
47           // #1 BŪDAS 
48           // Sukuriamas Person objekto sąrašas į kurį bus talpinami šio tipo masyvai 
49           List<Person[]> persons = new ArrayList<>(); 
50           // Idedamas į sąrašą Person objekto masyvas 
51           persons.add(person); 
52           // Kadangi sąraše yra masyvas duomenims pasiekti prireikia dvigubo ciklo 
53           // Pirmasis ciklas pereina visus sąrašo elementus 
54           // Antrasis ciklas pereina visus sąraše esančius masyvo person elementus 
55    
56           for(Person [] p : persons){ 
57               for(int i=0; i < p.length; i++){ 
58                   System.out.println("id:"+i+p[i]); 
59               } 
60           } 
61       } 
62    
63       void pushArrayElementsToList(Person [] person){ 
64    
65           // #2 BŪDAS 
66           // Turime sąrašą zmones į kurį bus dedami Person tipo objektų egzemplioriai 
67           List<Person> zmones = new ArrayList<>(); 
68    
69           // Iš masyvo person išrenkame visus elementus ir patalpiname sąraše žmones 
70           for(int j =0; j < person.length; j++){ 
71               zmones.add(person[j]); 
72           } 
73           // Duomenų išvedimas konsolėje sąrašo zmones 
74           for(Person p: zmones){ 
75               System.out.println(""+p); 
76           } 
77       } 
78       void getPersonAddress(Person [] person){ 
79           List<Person> zmones = new ArrayList<>(); 
80           // Iš masyvo person išrenkame visus elementus ir patalpiname sąraše žmones 
81           for(int j =0; j < person.length; j++){ 
82               zmones.add(person[j]); 
83           } 
84           // Duomenų išvedimas konsolėje sąrašo zmones 
85           for(Person p: zmones){ 
86               System.out.println(""+p.getAddress()); 
87           } 
88       }
