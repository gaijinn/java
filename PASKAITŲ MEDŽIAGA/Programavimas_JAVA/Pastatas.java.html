public class Pastatas { 
 
    String pavadinimas; 
    int aukstis; 
    int metai; 
 
    public Pastatas(){;} 
    public Pastatas(String pavadinimas, int aukstis, int metai) { 
        this.pavadinimas = pavadinimas; 
        this.aukstis = aukstis; 
        this.metai = metai; 
    } 
    public String getPavadinimas() { 
        return pavadinimas; 
    } 
 
    public void setPavadinimas(String pavadinimas) { 
        this.pavadinimas = pavadinimas; 
    } 
 
    public int getAukstis() { 
        return aukstis; 
    } 
 
    public void setAukstis(int aukstis) { 
        this.aukstis = aukstis; 
    } 
 
    public int getMetai() { 
        return metai; 
    } 
 
    public void setMetai(int metai) { 
        this.metai = metai; 
    } 
} 
class Daugiabutis extends Pastatas implements Paskirtis{ 
 
    String paskirtis; 
    public Daugiabutis(String pavadinimas, int aukstis, int metai) { 
        super(pavadinimas, aukstis, metai); 
    } 
 
    @Override 
    public void pridetiPaskirti(String paskirtis) { 
        this.paskirtis = paskirtis; 
    } 
}; 
class PrekybosCentras extends Pastatas{ 
    public PrekybosCentras(String pavadinimas, int aukstis, int metai) { 
        super(pavadinimas, aukstis, metai); 
    } 
}; 
class Main{ 
  public static void main(String [] args){ 
      Pastatas pastatas = new Daugiabutis("Daugiabutis", 20, 2015); 
 
      int aukstis = pastatas.getAukstis(); 
      int metai = pastatas.getMetai(); 
      String pavadinimas =  pastatas.getPavadinimas(); 
 
      System.out.println(aukstis +" "+ metai +" "+ pavadinimas); 
  } 
}; 
 
interface Paskirtis{ 
    void pridetiPaskirti(String paskirtis); 
};
